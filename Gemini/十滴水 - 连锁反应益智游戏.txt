    <script>
        const GRID_SIZE = 6;
        const boardElement = document.getElementById('game-board');
        const particleContainer = document.getElementById('particle-container');
        const dropsDisplay = document.getElementById('drops-count');
        const levelDisplay = document.getElementById('level-display');
        const modal = document.getElementById('modal');
        const modalTitle = document.getElementById('modal-title');
        const modalMsg = document.getElementById('modal-msg');
        const modalBtn = document.getElementById('modal-btn');
        const resetBtn = document.getElementById('reset-btn');

        let gameState = {
            grid: [], // 存储水滴等级 0-4
            dropsLeft: 10,
            level: 1,
            isAnimating: false,
            projectiles: []
        };

        // 初始化游戏
        function initGame(level = 1) {
            gameState.grid = Array(GRID_SIZE).fill().map(() => Array(GRID_SIZE).fill(0));
            gameState.dropsLeft = 10;
            gameState.level = level;
            gameState.isAnimating = false;
            
            // 随机生成初始水滴 (根据关卡增加难度)
            const initialCount = 12 + Math.min(level * 2, 15);
            for(let i = 0; i < initialCount; i++) {
                let r, c;
                do {
                    r = Math.floor(Math.random() * GRID_SIZE);
                    c = Math.floor(Math.random() * GRID_SIZE);
                } while (gameState.grid[r][c] !== 0);
                gameState.grid[r][c] = Math.floor(Math.random() * 4) + 1;
            }

            renderBoard();
            updateStats();
        }

        // 渲染棋盘
        function renderBoard() {
            boardElement.innerHTML = '';
            for(let r = 0; r < GRID_SIZE; r++) {
                for(let c = 0; c < GRID_SIZE; c++) {
                    const cell = document.createElement('div');
                    cell.className = 'cell';
                    cell.dataset.r = r;
                    cell.dataset.c = c;
                    cell.onclick = () => handleCellClick(r, c);

                    const level = gameState.grid[r][c];
                    if (level > 0) {
                        const drop = document.createElement('div');
                        drop.className = `drop drop-${level}`;
                        cell.appendChild(drop);
                    }
                    boardElement.appendChild(cell);
                }
            }
        }

        // 更新状态显示
        function updateStats() {
            dropsDisplay.innerText = gameState.dropsLeft;
            levelDisplay.innerText = gameState.level;
            
            if (gameState.dropsLeft <= 0 && !gameState.isAnimating && !checkWin()) {
                setTimeout(() => showModal('游戏结束', '你用完了所有水滴。', '再试一次', () => initGame(1)), 500);
            }
        }

        // 检查是否获胜
        function checkWin() {
            return gameState.grid.every(row => row.every(cell => cell === 0));
        }

        // 处理点击
        function handleCellClick(r, c) {
            if (gameState.isAnimating || gameState.dropsLeft <= 0) return;

            gameState.dropsLeft--;
            updateStats();
            increaseDrop(r, c);
        }

        // 增加水滴等级
        function increaseDrop(r, c) {
            if (r < 0 || r >= GRID_SIZE || c < 0 || c >= GRID_SIZE) return;

            gameState.grid[r][c]++;
            if (gameState.grid[r][c] > 4) {
                gameState.grid[r][c] = 0;
                burst(r, c);
            }
            renderBoard();
        }

        // 水滴爆裂
        async function burst(r, c) {
            gameState.isAnimating = true;
            
            const directions = [
                { dr: -1, dc: 0 }, // 上
                { dr: 1, dc: 0 },  // 下
                { dr: 0, dc: -1 }, // 左
                { dr: 0, dc: 1 }   // 右
            ];

            const projectiles = directions.map(dir => ({
                r, c,
                dr: dir.dr,
                dc: dir.dc,
                element: createProjectileElement(r, c)
            }));

            moveProjectiles(projectiles);
        }

        function createProjectileElement(r, c) {
            const el = document.createElement('div');
            el.className = 'projectile';
            const rect = boardElement.children[r * GRID_SIZE + c].getBoundingClientRect();
            const boardRect = boardElement.getBoundingClientRect();
            
            el.style.top = `${rect.top - boardRect.top + rect.height/2 - 6}px`;
            el.style.left = `${rect.left - boardRect.left + rect.width/2 - 6}px`;
            
            particleContainer.appendChild(el);
            return el;
        }

        function moveProjectiles(activeProjectiles) {
            if (activeProjectiles.length === 0) {
                gameState.isAnimating = false;
                if (checkWin()) {
                    setTimeout(() => showModal('干得漂亮！', `你通过了第 ${gameState.level} 关。`, '下一关', () => initGame(gameState.level + 1)), 500);
                }
                updateStats();
                return;
            }

            const speed = 6; // 每次移动的像素
            const nextBatch = [];

            activeProjectiles.forEach(p => {
                let top = parseFloat(p.element.style.top);
                let left = parseFloat(p.element.style.left);
                
                top += p.dr * speed;
                left += p.dc * speed;
                
                p.element.style.top = `${top}px`;
                p.element.style.left = `${left}px`;

                // 检查是否碰撞
                const currentR = Math.floor((top + 6) / (boardElement.offsetHeight / GRID_SIZE));
                const currentC = Math.floor((left + 6) / (boardElement.offsetWidth / GRID_SIZE));

                // 边界检查
                if (top < -10 || top > boardElement.offsetHeight + 10 || 
                    left < -10 || left > boardElement.offsetWidth + 10) {
                    p.element.remove();
                    return;
                }

                // 碰撞检测逻辑：如果进入了一个新格子且该格子有水滴
                if (currentR >= 0 && currentR < GRID_SIZE && currentC >= 0 && currentC < GRID_SIZE) {
                    // 我们记录上次经过的格子，防止在同一个格子里重复触发
                    if ((currentR !== p.lastR || currentC !== p.lastC) && gameState.grid[currentR][currentC] > 0) {
                        p.element.remove();
                        increaseDrop(currentR, currentC);
                        return;
                    }
                    p.lastR = currentR;
                    p.lastC = currentC;
                }

                nextBatch.push(p);
            });

            requestAnimationFrame(() => moveProjectiles(nextBatch));
        }

        // 弹窗控制
        function showModal(title, msg, btnText, callback) {
            modalTitle.innerText = title;
            modalMsg.innerText = msg;
            modalBtn.innerText = btnText;
            modal.classList.remove('hidden');
            
            modalBtn.onclick = () => {
                modal.classList.add('hidden');
                callback();
            };
        }

        resetBtn.onclick = () => initGame(1);

        // 启动游戏
        window.onload = () => initGame(1);
    </script>
