import React, { useState, useEffect, useRef } from 'react';
import { Upload, FileText, Image as ImageIcon, Loader2, Trash2, Copy, Check } from 'lucide-react';

const apiKey = ""; // 執行環境會自動提供 key

const App = () => {
  const [images, setImages] = useState([]); // { id, file, base64, status, text }
  const [activeId, setActiveId] = useState(null);
  const [isProcessing, setIsProcessing] = useState(false);
  const [copySuccess, setCopySuccess] = useState(false);
  const fileInputRef = useRef(null);

  // 指令碼定義
  const SYSTEM_PROMPT = "你是一個專業的 OCR 助手。請辨識圖片中的所有文字，並以結構化且易讀的方式呈現。不要加入額外的評論，只需輸出辨識出的文字內容。";

  // 指數退避請求函數
  const fetchWithRetry = async (payload, retries = 5, delay = 1000) => {
    for (let i = 0; i < retries; i++) {
      try {
        const response = await fetch(
          `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`,
          {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify(payload),
          }
        );
        if (response.ok) return await response.json();
        if (response.status !== 429 && i < retries - 1) throw new Error('API Error');
      } catch (e) {
        if (i === retries - 1) throw e;
      }
      await new Promise((res) => setTimeout(res, delay * Math.pow(2, i)));
    }
  };

  const handleFileChange = async (e) => {
    const files = Array.from(e.target.target?.files || e.target.files);
    if (!files.length) return;

    const newImages = await Promise.all(
      files.map(async (file) => {
        const base64 = await fileToBase64(file);
        return {
          id: Math.random().toString(36).substr(2, 9),
          file,
          base64: base64.split(',')[1],
          preview: URL.createObjectURL(file),
          status: 'pending',
          text: ''
        };
      })
    );

    setImages(prev => [...prev, ...newImages]);
    if (!activeId && newImages.length > 0) setActiveId(newImages[0].id);
  };

  const fileToBase64 = (file) => {
    return new Promise((resolve, reject) => {
      const reader = new FileReader();
      reader.readAsDataURL(file);
      reader.onload = () => resolve(reader.result);
      reader.onerror = (error) => reject(error);
    });
  };

  const runOCR = async (id) => {
    const target = images.find(img => img.id === id);
    if (!target || target.status === 'processing') return;

    // 更新狀態為處理中
    setImages(prev => prev.map(img => img.id === id ? { ...img, status: 'processing' } : img));
    setIsProcessing(true);

    const payload = {
      contents: [{
        parts: [
          { text: "請辨識這張圖片中的文字。" },
          { inlineData: { mimeType: "image/png", data: target.base64 } }
        ]
      }],
      systemInstruction: { parts: [{ text: SYSTEM_PROMPT }] }
    };

    try {
      const result = await fetchWithRetry(payload);
      const extractedText = result.candidates?.[0]?.content?.parts?.[0]?.text || "未能辨識出文字。";
      
      setImages(prev => prev.map(img => 
        img.id === id ? { ...img, status: 'completed', text: extractedText } : img
      ));
    } catch (error) {
      setImages(prev => prev.map(img => 
        img.id === id ? { ...img, status: 'error', text: "辨識失敗，請重試。" } : img
      ));
    } finally {
      setIsProcessing(false);
    }
  };

  const removeImage = (id, e) => {
    e.stopPropagation();
    setImages(prev => prev.filter(img => img.id !== id));
    if (activeId === id) setActiveId(null);
  };

  const activeImage = images.find(img => img.id === activeId);

  const copyToClipboard = () => {
    if (!activeImage?.text) return;
    const textArea = document.createElement("textarea");
    textArea.value = activeImage.text;
    document.body.appendChild(textArea);
    textArea.select();
    try {
      document.execCommand('copy');
      setCopySuccess(true);
      setTimeout(() => setCopySuccess(false), 2000);
    } catch (err) {
      console.error('無法複製文字', err);
    }
    document.body.removeChild(textArea);
  };

  return (
    <div className="flex flex-col h-screen bg-slate-50 text-slate-900 font-sans">
      {/* Header */}
      <header className="bg-white border-b px-6 py-4 flex justify-between items-center shrink-0">
        <div className="flex items-center gap-2">
          <div className="bg-blue-600 p-2 rounded-lg">
            <FileText className="text-white w-5 h-5" />
          </div>
          <h1 className="text-xl font-bold tracking-tight">AI 多圖片 OCR 辨識系統</h1>
        </div>
        <button 
          onClick={() => fileInputRef.current.click()}
          className="flex items-center gap-2 bg-blue-600 hover:bg-blue-700 text-white px-4 py-2 rounded-full transition-all shadow-sm"
        >
          <Upload size={18} />
          <span>上傳圖片</span>
        </button>
        <input 
          type="file" 
          ref={fileInputRef} 
          onChange={handleFileChange} 
          multiple 
          accept="image/*" 
          className="hidden" 
        />
      </header>

      {/* Main Content */}
      <main className="flex flex-1 overflow-hidden">
        {/* Left Side: Image List */}
        <section className="w-1/3 border-r bg-white flex flex-col overflow-hidden">
          <div className="p-4 border-b bg-slate-50/50 flex justify-between items-center">
            <h2 className="text-sm font-semibold text-slate-500 uppercase tracking-wider">待處理清單 ({images.length})</h2>
          </div>
          <div className="flex-1 overflow-y-auto p-4 space-y-3">
            {images.length === 0 ? (
              <div className="h-full flex flex-col items-center justify-center text-slate-400 opacity-60 italic">
                <ImageIcon size={48} className="mb-2" />
                <p>尚未上傳任何圖片</p>
              </div>
            ) : (
              images.map((img) => (
                <div 
                  key={img.id}
                  onClick={() => setActiveId(img.id)}
                  className={`group relative p-3 rounded-xl border-2 cursor-pointer transition-all ${
                    activeId === img.id ? 'border-blue-500 bg-blue-50' : 'border-transparent bg-slate-100 hover:bg-slate-200'
                  }`}
                >
                  <div className="flex gap-3 items-center">
                    <img src={img.preview} alt="preview" className="w-16 h-16 object-cover rounded-lg shadow-sm" />
                    <div className="flex-1 min-w-0">
                      <p className="text-sm font-medium truncate">{img.file.name}</p>
                      <div className="mt-1">
                        {img.status === 'pending' && (
                          <button 
                            onClick={(e) => { e.stopPropagation(); runOCR(img.id); }}
                            className="text-xs text-blue-600 font-semibold hover:underline"
                          >
                            開始辨識
                          </button>
                        )}
                        {img.status === 'processing' && (
                          <span className="text-xs text-amber-600 flex items-center gap-1">
                            <Loader2 size={12} className="animate-spin" /> 辨識中...
                          </span>
                        )}
                        {img.status === 'completed' && (
                          <span className="text-xs text-green-600">辨識完成</span>
                        )}
                        {img.status === 'error' && (
                          <span className="text-xs text-red-600">辨識失敗</span>
                        )}
                      </div>
                    </div>
                  </div>
                  <button 
                    onClick={(e) => removeImage(img.id, e)}
                    className="absolute -top-2 -right-2 p-1 bg-white border shadow-md rounded-full text-slate-400 hover:text-red-500 opacity-0 group-hover:opacity-100 transition-opacity"
                  >
                    <Trash2 size={14} />
                  </button>
                </div>
              ))
            )}
          </div>
        </section>

        {/* Right Side: Text Output */}
        <section className="flex-1 flex flex-col bg-slate-50 overflow-hidden relative">
          {!activeImage ? (
            <div className="flex-1 flex flex-col items-center justify-center text-slate-400">
              <p>選擇左側圖片以查看結果</p>
            </div>
          ) : (
            <>
              <div className="p-4 bg-white border-b flex justify-between items-center shadow-sm z-10">
                <div className="flex flex-col">
                  <h3 className="font-bold text-slate-700">辨識結果</h3>
                  <span className="text-xs text-slate-400">{activeImage.file.name}</span>
                </div>
                <div className="flex items-center gap-2">
                  <button 
                    onClick={copyToClipboard}
                    disabled={!activeImage.text}
                    className="flex items-center gap-1.5 px-3 py-1.5 rounded-lg text-sm bg-slate-100 hover:bg-slate-200 text-slate-700 disabled:opacity-50 transition-colors"
                  >
                    {copySuccess ? <Check size={16} className="text-green-600" /> : <Copy size={16} />}
                    {copySuccess ? '已複製' : '複製文字'}
                  </button>
                </div>
              </div>
              
              <div className="flex-1 overflow-y-auto p-8">
                <div className="max-w-3xl mx-auto space-y-8">
                  {/* Image Preview Area in detail */}
                  <div className="bg-white p-4 rounded-2xl shadow-sm border">
                    <img 
                      src={activeImage.preview} 
                      className="w-full max-h-80 object-contain rounded-lg" 
                      alt="Full View" 
                    />
                  </div>

                  {/* Text Content Area */}
                  <div className="bg-white p-8 rounded-2xl shadow-md border min-h-[400px] relative">
                    {activeImage.status === 'processing' ? (
                      <div className="absolute inset-0 flex flex-col items-center justify-center bg-white/80 rounded-2xl">
                        <Loader2 className="animate-spin text-blue-600 mb-2" size={32} />
                        <p className="text-slate-500 font-medium text-lg">正在解析圖片內容...</p>
                      </div>
                    ) : activeImage.status === 'pending' ? (
                      <div className="h-full flex flex-col items-center justify-center text-slate-400 py-20 border-2 border-dashed border-slate-200 rounded-xl">
                        <p className="mb-4">尚未開始辨識</p>
                        <button 
                          onClick={() => runOCR(activeImage.id)}
                          className="px-6 py-2 bg-blue-600 text-white rounded-full hover:bg-blue-700 shadow-md"
                        >
                          立即辨識
                        </button>
                      </div>
                    ) : (
                      <div className="whitespace-pre-wrap text-slate-800 leading-relaxed font-mono">
                        {activeImage.text}
                      </div>
                    )}
                  </div>
                </div>
              </div>
            </>
          )}
        </section>
      </main>
    </div>
  );
};

export default App;
