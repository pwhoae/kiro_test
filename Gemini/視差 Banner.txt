    <script>
        const banner = document.getElementById('parallaxBanner');
        const urlInput = document.getElementById('imageUrlInput');
        const fileInput = document.getElementById('fileInput');

        // 滑鼠互動邏輯
        banner.addEventListener('mousemove', (e) => {
            const { left, top, width, height } = banner.getBoundingClientRect();
            const x = (e.clientX - left) / width;
            const y = (e.clientY - top) / height;
            
            // 計算旋轉角度 (限制在 -5deg 到 5deg 之間)
            const moveX = (x - 0.5) * 10; 
            const moveY = (y - 0.5) * -10;

            document.documentElement.style.setProperty('--m-x', `${moveX}deg`);
            document.documentElement.style.setProperty('--m-y', `${moveY}deg`);
        });

        // 離開時恢復原狀
        banner.addEventListener('mouseleave', () => {
            document.documentElement.style.setProperty('--m-x', `0deg`);
            document.documentElement.style.setProperty('--m-y', `0deg`);
        });

        // 圖片處理邏輯
        function setBackgroundImage(src) {
            document.documentElement.style.setProperty('--parallax-img', `url('${src}')`);
        }

        function updateFromUrl() {
            if (urlInput.value) setBackgroundImage(urlInput.value);
        }

        fileInput.addEventListener('change', (e) => {
            const file = e.target.files[0];
            if (file) {
                const reader = new FileReader();
                reader.onload = (ev) => setBackgroundImage(ev.target.result);
                reader.readAsDataURL(file);
            }
        });
    </script>
